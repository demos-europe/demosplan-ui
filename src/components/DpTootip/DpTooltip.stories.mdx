import { Canvas, Meta, Story } from '@storybook/addon-docs'
import { DpButton } from '../index'
import DpTooltip from './DpTooltip.vue'
import { Tooltip } from '../../directives'

<Meta
argTypes={{
  default: {
    text: {
      type: 'text'
    }
  }
}}
parameters={{
  slots: {
    default: {
      description: 'Default slot',
      template: `<p>{{ args.default }}</p>`,
    },
  }
}}
title="Components/Tooltip"
component={DpTooltip} />

export const Template = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { DpTooltip },
  template: `<dp-tooltip v-bind="propsWithoutDefaultSlot" >{{ slotcontent }}</dp-tooltip>`,
  computed: {
    propsWithoutDefaultSlot() {
      const all = this.$props
      delete all['default']
      return all
    }
  }
});


export const DirectiveTemplate = (args, { argTypes }) => ({
    props: Object.keys(argTypes),
    components: { DpButton },
    template: `<DpButton v-tooltip="'${Object.values(args)[0]}'">My button</DpButton>`,
    directives: {
      tooltip: Tooltip
    },
    computed: {
      propsWithoutDefaultSlot () {
        const all = this.$props
        delete all['default']
        return all
      }
    }
})


# Tooltip

Renders a Tooltip.


## Usage without HTML

```html
<dp-tooltip text="Content of the tooltip">
  slot content
</dp-tooltip>
```

## Usage with HTML

```html
<dp-tooltip text="Content of <strong>the Tooltip </strong>">
  slot content
</dp-tooltip>
```

## Usage as Directive

```html
<my-container v-tooltip="Content of <strong>the Tooltip </strong>" />
```
